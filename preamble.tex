{\begin{itemize}%

	\setlength{\parskip}{0pt}}%
{\end{itemize}}

{\begin{enumerate}%
	\setlength{\itemsep}{0pt}%
	\setlength{\parskip}{0pt}}%
{\end{enumerate}}
